Skrypt SQL do ćwiczeń

Skrypt możesz skopiować do pliku cwiczenia.sql i wykonywać po kawałku w konsoli MySQL. 
-- ============================================
--  OBSŁUGA KONSOLI MySQL (XAMPP)
-- ============================================

-- Logowanie jako root (bez hasła w XAMPP):
mysql -u root

-- Pokaż dostępne bazy danych:
SHOW DATABASES;

-- Użyj wybranej bazy danych:
USE <nazwa_bazy>;

-- Pokaż strukturę tabeli:
DESCRIBE <nazwa_tabeli>;


-- ============================================
--  PRZYKŁAD: TABELA osoba
-- ============================================

CREATE TABLE osoba (
    id_osoby INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    nazwisko VARCHAR(30) NOT NULL,
    rok_zameldowania VARCHAR(4) DEFAULT '2004',
    zamieszkanie VARCHAR(20) DEFAULT 'Dzialdowo'
);

-- UWAGA:
-- AUTO_INCREMENT = numerowanie identyfikatora od 1
-- Brak AUTO_INCREMENT → możesz np. wpisać id = 0


-- ============================================
--  MODYFIKACJA ISTNIEJĄCEJ TABELI osoba
-- ============================================

-- DODAWANIE NOWEJ KOLUMNY:
ALTER TABLE osoba ADD pesel INT NOT NULL;

-- MODYFIKACJA KOLUMNY (zmiana długości nazwiska):
ALTER TABLE osoba
MODIFY COLUMN nazwisko VARCHAR(40) NOT NULL;

-- MODYFIKACJA KOLUMNY ZE ZMIANĄ NAZWY:
ALTER TABLE osoba
CHANGE COLUMN nazwisko nazwisko2 VARCHAR(35) NOT NULL;

-- USUWANIE KOLUMNY:
ALTER TABLE osoba
DROP COLUMN pesel;


-- ============================================
--  OGRANICZENIA (CHECK) I WPROWADZANIE DANYCH
-- ============================================

-- Dodanie kolumny z ograniczeniem wiek 18–90:
ALTER TABLE osoba
ADD wiek INT CHECK (wiek BETWEEN 18 AND 90);

-- Wprowadzanie danych (INSERT):
-- INSERT INTO nazwa_tabeli(kol1, kol2, kol3)
-- VALUES (wart1, wart2, wart3);

-- Przykład:
INSERT INTO osoba(nazwisko2, rok_zameldowania, zamieszkanie, wiek)
VALUES ('Kowalski', '2010', 'Dzialdowo', 25);

-- Sprawdzenie danych:
SELECT * FROM osoba;

-- Aktualizacja danych (UPDATE):
UPDATE osoba
SET imie = 'Andrzej'
WHERE id_osoby = 1;

-- ============================================
--  ZAPYTANIA DO TABELI pracownik
--  (tu zakładamy tabelę: pracownik)
-- ============================================

-- Przykładowa struktura (dopasuj do swojej bazy):
CREATE TABLE pracownik (
    id_pracownika INT PRIMARY KEY AUTO_INCREMENT,
    imie VARCHAR(20),
    nazwisko VARCHAR(30),
    zawod VARCHAR(20),
    wynagrodzenie DECIMAL(7,2),
    premia DECIMAL(7,2),
    wydzial INT,
    data_urodzenia DATE
);

-- Przykładowe zapytania:

-- 1. Wyświetl wszystkie rekordy:
SELECT * FROM pracownik;

-- 2. Wszystkie nazwiska:
SELECT nazwisko FROM pracownik;

-- 3. Nazwiska oraz wynagrodzenia:
SELECT nazwisko, wynagrodzenie FROM pracownik;

-- 4. Jakie zawody wykonują pracownicy:
SELECT DISTINCT zawod FROM pracownik;


-- ============================================
--  FUNKCJE AGREGUJĄCE (COUNT, SUM, AVG, MIN, MAX)
-- ============================================

-- 1) Ilu jest pracowników?
SELECT COUNT(*) AS liczba_pracownikow
FROM pracownik;

-- 2) Ilu ma wpisaną premię? (NULL nie jest liczony)
SELECT COUNT(premia) AS z_premia
FROM pracownik;

-- 3) Ile jest różnych wydziałów?
SELECT COUNT(DISTINCT wydzial) AS liczba_wydzialow
FROM pracownik;

-- 4) Średnie wynagrodzenie:
SELECT ROUND(AVG(wynagrodzenie), 2) AS srednie_wynagrodzenie
FROM pracownik;

-- 5) Najniższe i najwyższe wynagrodzenie:
SELECT
    MIN(wynagrodzenie) AS min_wyn,
    MAX(wynagrodzenie) AS max_wyn
FROM pracownik;

-- 6) Suma wynagrodzeń:
SELECT SUM(wynagrodzenie) AS suma_wynagrodzen
FROM pracownik;

-- 7) Suma premii (tylko nie-NULL):
SELECT SUM(premia) AS suma_premii
FROM pracownik;

-- 8) Średnia premia (tylko dla tych, co ją mają):
SELECT ROUND(AVG(premia), 2) AS srednia_premia
FROM pracownik;
